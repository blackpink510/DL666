def train_one_epoch(epoch, model, train_loader, criterion, optimizer, device, log_step=22):
    model.train()
    running_loss = 0.0
    running_loop = 0.0
    
    for i, (inputs, labels) in enumerate(train_loader):
        
        inputs, labels = inputs.to(device), labels.to(device)
        
        optimizer.zero_grad()
        outputs = model(inputs)
        loss = criterion(outputs, labels)
        loss.backward()
        optimizer.step()

        # train loss is the averaged over multiple iterations
        running_loss += loss.item()
        running_loop += 1
        train_loss = running_loss / running_loop
        
        if i % log_step == log_step - 1 or i == len(train_loader) - 1:                
            print(f'[Epoch {epoch + 1:2d}, iter {i + 1:5d}] train_loss: {train_loss:.3f}')
            running_loss = 0.0
            running_loop = 0.0
    
    return train_loss

def evaluate(model, test_loader, criterion, device):
    model.eval()
    correct = 0
    total = 0
    test_loss = 0.0
    with torch.inference_mode():
        for inputs, labels in test_loader:
            inputs, labels = inputs.to(device), labels.to(device)
            outputs = model(inputs)
            loss = criterion(outputs, labels)
            test_loss += loss.item()
            _, predicted = torch.max(outputs.data, 1)
            total += labels.size(0)
            correct += (predicted == labels).sum().item()
    
    test_loss = test_loss / len(test_loader.dataset)
    test_acc = 100 * correct / total
        
    return test_loss, test_acc

def train(model, optimizer, criterion, train_loader, test_loader, config, device):    
    
    for epoch in range(config['num_epochs']):
        train_loss = train_one_epoch(epoch, model, train_loader, criterion, optimizer, device)
        test_loss, test_acc = evaluate(model, test_loader, criterion, device)
        
        print(f"--> Epoch {epoch+1} completed. Train Loss: {train_loss:.3f}, Test Loss: {test_loss:.3f}, Test Accuracy: {test_acc:.2f}%\n")

class ChestXRayDataset(Dataset):
    def __init__(self, path, transform=None):
        self.path = path
        self.transform = transform
        self.classes = os.listdir(path)
        self.targets = []
        self.images_path = []

        for i, cls in enumerate(classes):
            cls_path = os.path.join(path, cls)
            cls_samples = [os.path.join(cls_path, filename) for filename in os.listdir(cls_path)]
            self.images_path += cls_samples
            self.targets += [i]*len(cls_samples)

    def __len__(self):
        return len(self.targets)
        

    def __getitem__(self, idx):
        filename = os.path.join(self.path, self.images_path[idx])
        x = Image.open(filename).convert("RGB")
        if self.transform is not None:
            x = self.transform(x)
        y = self.targets[idx]
        
        return x, y

def build_model(num_classes, config, device=torch.device("cpu"), return_optimizer=True):

    # load the pretrained model 
    model = resnet18(weights=config['weights']).to(device)

    if config['weights'] is None:
        # Training from scratch - enable all layers 
        pass
    elif config['finetuned_layers'] == 'output':
        # Use pretrained model as a feature extractor - disable all layers
        for param in model.parameters():
            param.requires_grad = False        
    elif config['finetuned_layers'] == 'all':
        # Finetune all layers - enable all layers
        pass
    else:
        # freeze the layers that are not listed in config['finetuned_layers']
        for name, layer in model.named_parameters():
            if not any([name.startswith(layer_name) for layer_name in config['finetuned_layers']]):
                layer.requires_grad = False

    # replace the classification layer
    model.fc = nn.Linear(model.fc.in_features, num_classes).to(device)

    # transfer model to GPU if available
    model = model.to(device)

    # return model only
    if not return_optimizer:
        return model

    # create the optimizer
    optimizer = optim.SGD(model.parameters(), lr=config['lr'], momentum=config['momentum'])

    # return model and optimizer
    return model, optimizer

def plot_predictions(model, test_loader, device):
    model.eval()
    with torch.no_grad():
        inputs, labels = next(iter(test_loader))
        inputs, labels = inputs.to(device), labels.to(device)
        outputs = model(inputs)
        _, predicted = torch.max(outputs, 1)
        
        # Plot the images with their predicted labels
        plt.figure(figsize=(10, 10))
        for i in range(9):
            plt.subplot(3, 3, i+1)
            img2 = inputs[i].cpu()
            img = np.transpose(inputs[i].cpu().numpy(), (1, 2, 0))*0.29 + 0.5 # undo normalization
            img = np.clip(img, 0, 1)
            plt.imshow(img)
            plt.title(f"Pred: {dataset.classes[predicted[i].item()]}, Actual: {dataset.classes[labels[i].item()]}", fontsize=10)
            plt.axis('off')
        plt.show()

Section 8: Saving the Model¶

Save the state dictionary (parameters and variables) of model3 into a file named "final_model.safetensors". The save_file function ensures that the model's weights are stored in the .safetensors format, which is a more secure and efficient alternative to traditional PyTorch .pt or .pth files. This allows for safe and deterministic loading of the model later for inference or further training.

In [54]:
model_path = "final_model.safetensors"
save_file(model3.state_dict(), model_path)

You can list the files in the working directory to view the saved model. You can download the model file by navigating to the Notebook toolbar on the right, expand the Output tab and right click on the saved model to download the model.

Section 8: Load a Model¶

Load the saved state dictionary into loaded_state_dict. The load_file function retrieves the model parameters stored in the safetensors format, which can then be used to initialize a model for inference or further training. However, note that this only loads the state dictionary — you still need to load it into a model instance using model.load_state_dict(loaded_state_dict).

In [56]:
model_path = "final_model.safetensors"
loaded_state_dict = load_file(model_path)
